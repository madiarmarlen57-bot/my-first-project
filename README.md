num1 = int(input())
num3 = input()
num2 = int(input())
if num3 == "+":
  print(num1 + num2)
elif num3 == "-":
  print(num1 - num2)
elif num3 == "*":
  print(num1 * num2)
elif num3 == "/":
  print(num1 / num2)
else:
  print('Error')
